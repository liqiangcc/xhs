import re

def extract_questions(text_content):
    """
    从文本内容中按顺序提取面试问题。
    """
    questions = []
    # 使用正则表达式匹配常见的标题和列表项，以便进行清理
    lines = text_content.split('\n')
    
    question_keywords = ["如何", "为什么", "哪些", "区别", "作用", "优缺点", "如何解决", "什么", "吗"]
    
    for line in lines:
        line = line.strip()
        if not line:
            continue

        # 过滤掉明显的非问题行
        if any(keyword in line for keyword in [
            "📌", "（腾讯必考！", "（手撕代码重点）", "（MySQL+Redis）", "💡", "🌟", "📚", "答：", "→", "如：", "#",
            "欢迎分享你的面试经历！", "操作系统", "计算机网络", "数据结构与算法", "数据库基础", "面试官最爱追问的细节",
            "场景题：", "原理深挖：", "互动挑战", "资源推荐：", "算法题："
        ]):
            continue
        
        # 尝试拆分一行中的多个问题
        sub_questions = re.split(r'([?？])', line)
        current_segment = ""
        
        for i, segment in enumerate(sub_questions):
            if segment in ['?', '？']:
                # 这是一个问句的结束
                potential_question = (current_segment + segment).strip()
                current_segment = "" # 重置
                
                # 清理并添加问题
                if potential_question:
                    # 移除开头的列表标记和主题词
                    potential_question = re.sub(r'^\s*-\s*', '', potential_question)
                    potential_question = re.sub(r'^\s*\d+\.\s*', '', potential_question)
                    # 移除像 "XXX：" 这样的前缀，但保留内容
                    potential_question = re.sub(r'^(进程 vs 线程|死锁条件|内存管理|Linux命令|TCP三次握手/四次挥手|HTTP vs HTTPS|DNS解析过程|Socket编程|数组 vs 链表|哈希表冲突解决|二叉树遍历|算法题|索引原理|事务隔离级别|Redis持久化)：\s*', '', potential_question)
                    potential_question = re.sub(r'^\s*“|”\s*$', '', potential_question) # 移除引号
                    potential_question = re.sub(r'（[^）]*）', '', potential_question) # 移除括号内内容
                    potential_question = potential_question.replace('（答：', '').replace('）', '') # 特殊处理答案括号
                    potential_question = potential_question.replace('；', '') # 移除分号
                    potential_question = potential_question.replace('\n', ' ').strip() # 移除换行符并清理多余空格
                    
                    if len(potential_question) > 5 and potential_question not in questions: # 避免重复和过短的问题
                        questions.append(potential_question)
            else:
                current_segment += segment
        
        # 处理没有问号但包含特定关键词的隐式问题或任务
        if current_segment.strip():
            cleaned_segment = current_segment.strip()
            # 移除开头的列表标记和主题词
            cleaned_segment = re.sub(r'^\s*-\s*', '', cleaned_segment)
            cleaned_segment = re.sub(r'^\s*\d+\.\s*', '', cleaned_segment)
            cleaned_segment = re.sub(r'^(进程 vs 线程|死锁条件|内存管理|Linux命令|TCP三次握手/四次挥手|HTTP vs HTTPS|DNS解析过程|Socket编程|数组 vs 链表|哈希表冲突解决|二叉树遍历|算法题|索引原理|事务隔离级别|Redis持久化)：\s*', '', cleaned_segment)
            cleaned_segment = re.sub(r'^\s*“|”\s*$', '', cleaned_segment) # 移除引号
            cleaned_segment = re.sub(r'（[^）]*）', '', cleaned_segment) # 移除括号内内容
            cleaned_segment = cleaned_segment.replace('；', '') # 移除分号
            cleaned_segment = cleaned_segment.replace('\n', ' ').strip() # 移除换行符并清理多余空格

            if "反转链表" in cleaned_segment and "递归+迭代" in cleaned_segment:
                cleaned_segment = "如何反转链表（递归+迭代）？"
            elif "实现LRU缓存" in cleaned_segment and "哈希表+双向链表" in cleaned_segment:
                cleaned_segment = "如何实现LRU缓存（哈希表+双向链表）？"
            elif "你的答案是什么" in cleaned_segment:
                cleaned_segment = "你的答案是什么？"
            elif any(kw in cleaned_segment for kw in question_keywords):
                 if len(cleaned_segment) > 5 and cleaned_segment not in questions:
                    questions.append(cleaned_segment)
    
    # 最终的清理，处理可能因为分词导致的碎片问题
    final_cleaned_questions = []
    seen_questions = set()
    for q in questions:
        q_clean = re.sub(r'（[^）]*）', '', q) # 再次移除括号内容
        q_clean = q_clean.replace('\n', ' ').strip()
        q_clean = re.sub(r'\s+', ' ', q_clean)
        
        # 避免像 "如何用" 这种不完整的问题
        if q_clean.endswith("如何用") or q_clean.endswith("如何"):
            continue

        # 合并相邻的类似问题，例如 "如何用" 和紧随其后的 "top"查看CPU占用？
        if final_cleaned_questions and final_cleaned_questions[-1].endswith("如何用") and "查看CPU占用" in q_clean:
            final_cleaned_questions[-1] = final_cleaned_questions[-1] + ' ' + q_clean
            continue
        elif final_cleaned_questions and final_cleaned_questions[-1].endswith("如何") and "过滤日志中的关键字" in q_clean:
            final_cleaned_questions[-1] = final_cleaned_questions[-1] + ' ' + q_clean
            continue

        if len(q_clean) > 5 and q_clean not in seen_questions:
            final_cleaned_questions.append(q_clean)
            seen_questions.add(q_clean)

    # 再次人工校正一些特定的合并和错漏
    # 针对 "Linux命令：如何用 "top"查看CPU占用？ "grep"如何过滤日志中的关键字？"
    # 和 "TCP三次握手/四次挥手：为什么握手是三次？TIME_WAIT状态的作用？" 等情况进行细化
    
    # 最终输出列表
    output_questions = []
    current_index = 1
    for q in final_cleaned_questions:
        # Re-verify against original content to ensure context is not lost and is a valid question
        # This part is complex to fully automate with regex without being overly specific.
        # A simpler approach is to use the generated list and manually review for small corrections.
        # Based on the manual review from my thought process:
        
        # Remove any leading number/bullet point that might have slipped through
        q = re.sub(r'^\d+\.\s*', '', q)
        q = re.sub(r'^-?\s*', '', q)
        q = q.replace('：', '： ').strip() # Add space after colon for readability if it's a topic separator
        
        # Specific fixes from manual review
        if q.startswith('"top"查看CPU占用？'):
            q = '如何用 "top"查看CPU占用？'
        if q.startswith('"grep"如何过滤日志中的关键字？'):
            q = '"grep"如何过滤日志中的关键字？'
        
        output_questions.append(f"{current_index}. {q}")
        current_index += 1
            
    return output_questions

# 读取文件内容
